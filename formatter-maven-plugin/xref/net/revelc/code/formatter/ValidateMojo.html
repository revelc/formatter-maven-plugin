17  package net.revelc.code.formatter;
18  
19  import java.io.File;
20  import java.io.IOException;
21  import java.util.Properties;
22  
23  import org.eclipse.jface.text.BadLocationException;
24  
25  import org.apache.maven.plugin.MojoExecutionException;
26  import org.apache.maven.plugin.MojoFailureException;
27  import org.apache.maven.plugins.annotations.LifecyclePhase;
28  import org.apache.maven.plugins.annotations.Mojo;
29  import org.apache.maven.plugins.annotations.Parameter;
30  
31  /**
32   * This mojo is very similar to Formatter mojo, but it is focused on CI servers.
33   * 
34   * If the code ain't formatted as expected this mojo will fail the build
35   * 
36   * @author marvin.froeder
37   */
38  @Mojo(name = "validate", defaultPhase = LifecyclePhase.VALIDATE, requiresProject = false)
39  public class ValidateMojo extends FormatterMojo {
40  
41      @Parameter(defaultValue = "false", property = "aggregator", required = true)
42      private boolean aggregator;
43  
44      @Parameter(defaultValue = "${project.executionRoot}", required = true)
45      private boolean executionRoot;
46  
47      @Override
48      public void execute() throws MojoExecutionException, MojoFailureException {
49          if (this.aggregator && !this.executionRoot) {
50              return;
51          }
52  
53          super.execute();
54      }
55  
56      @Override
57      protected void doFormatFile(File file, ResultCollector rc, Properties hashCache, String basedirPath, boolean dryRun)
58              throws IOException, MojoFailureException, BadLocationException, MojoExecutionException {
59          super.doFormatFile(file, rc, hashCache, basedirPath, true);
60  
61          if (rc.successCount != 0) {
62              throw new MojoFailureException("File '" + file + "' format doesn't match!");
63          }
64          if (rc.failCount != 0) {
65              throw new MojoExecutionException("Error formating '" + file + "' ");
66          }
67      }
68  
69  }
